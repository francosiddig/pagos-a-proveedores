Configuración inicial

Conectá la app con tu Google Sheet. Solo hay que hacerlo una vez.

  1. Abrí Google Sheets y creá una hoja nueva
  2. En la fila 1 poné estos encabezados:
    Fecha | Hora | Proveedor | Cobrador | Importe | Forma de Pago | Registrado por | Firma
  3. Andá a Extensiones → Apps Script, pegá el script de abajo, guardá
  4. Implementar → Nueva implementación → Tipo: App web → Acceso: Cualquiera → Implementar
  5. Copiá la URL y pegala abajo
▶ Ver script de Apps Script 
function doPost(e) {
  try {
    var data = JSON.parse(e.postData.contents);
    var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
    sheet.appendRow([
      data.fecha, data.hora, data.proveedor, data.cobrador,
      data.importe, data.formaPago, data.registradoPor,
      data.firma ? "✓ firmado" : "sin firma"
    ]);
    return ContentService
      .createTextOutput(JSON.stringify({ok: true}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService
      .createTextOutput(JSON.stringify({ok: false, error: err.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
